88       @Override 
89       public void onConnectionFailed(@NonNull ConnectionResult connectionResult) { 
90           Toast.makeText(this, "Connection Failed!", Toast.LENGTH_SHORT).show(); 
91           if (connectionResult.hasResolution()) { 
92               try { 
93                   // Start an Activity that tries to resolve the error 
94                   connectionResult.startResolutionForResult(this, 90000); 
95               } catch (IntentSender.SendIntentException e) { 
96                   e.printStackTrace(); 
97               } 
98           } else { 
99               Log.i("Current Location", "Location services connection failed with code " + connectionResult.getErrorCode()); 
100          }

104      public void settingRequest() { 
105          mLocationRequest = new LocationRequest(); 
106          mLocationRequest.setInterval(10000);    // 10 seconds, in milliseconds 
107          mLocationRequest.setFastestInterval(1000);   // 1 second, in milliseconds 
108          mLocationRequest.setPriority(LocationRequest.PRIORITY_HIGH_ACCURACY); 
109   
110          LocationSettingsRequest.Builder builder = new LocationSettingsRequest.Builder() 
111                  .addLocationRequest(mLocationRequest); 
112   
113          PendingResult<LocationSettingsResult> result = 
114                  LocationServices.SettingsApi.checkLocationSettings(mGoogleApiClient, 
115                          builder.build()); 
116   
117          result.setResultCallback(new ResultCallback<LocationSettingsResult>() { 
118   
119              @Override 
120              public void onResult(@NonNull LocationSettingsResult result) { 
121                  final Status status = result.getStatus(); 
122                  final LocationSettingsStates state = result.getLocationSettingsStates(); 
123                  switch (status.getStatusCode()) { 
124                      case LocationSettingsStatusCodes.SUCCESS: 
125                          // All location settings are satisfied. The client can 
126                          // initialize location requests here. 
127                          getLocation(); 
128                          break; 
129                      case LocationSettingsStatusCodes.RESOLUTION_REQUIRED: 
130                          // Location settings are not satisfied, but this can be fixed 
131                          // by showing the user a dialog. 
132                          try { 
133                              // Show the dialog by calling startResolutionForResult(), 
134                              // and check the result in onActivityResult(). 
135                              status.startResolutionForResult(MainActivity.this, 1000); 
136                          } catch (SendIntentException e) { 
137                              // Ignore the error. 
138                          } 
139                          break; 
140                      case LocationSettingsStatusCodes.SETTINGS_CHANGE_UNAVAILABLE: 
141                          // Location settings are not satisfied. However, we have no way 
142                          // to fix the settings so we won't show the dialog. 
143                          break; 
144                  } 
145              } 
146   
147          }); 
148      }

171      public void getLocation() { 
172          if (ActivityCompat.checkSelfPermission(this, android.Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED && ActivityCompat.checkSelfPermission(this, android.Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED) { 
173              // TODO: Consider calling 
174              //    ActivityCompat#requestPermissions 
175              // here to request the missing permissions, and then overriding 
176              //   public void onRequestPermissionsResult(int requestCode, String[] permissions, 
177              //                                          int[] grantResults) 
178              // to handle the case where the user grants the permission. See the documentation 
179              // for ActivityCompat#requestPermissions for more details. 
180              return; 
181          } else { 
182              /*Getting the location after aquiring location service*/ 
183              mLastLocation = LocationServices.FusedLocationApi.getLastLocation( 
184                      mGoogleApiClient); 
185   
186              if (mLastLocation != null) { 
187                  _progressBar.setVisibility(View.INVISIBLE); 
188                  _latitude.setText("Latitude: " + String.valueOf(mLastLocation.getLatitude())); 
189                  _longitude.setText("Longitude: " + String.valueOf(mLastLocation.getLongitude())); 
190              } else { 
191                  /*if there is no last known location. Which means the device has no data for the loction currently. 
192                  * So we will get the current location. 
193                  * For this we'll implement Location Listener and override onLocationChanged*/ 
194                  Log.i("Current Location", "No data for location found"); 
195   
196                  if (!mGoogleApiClient.isConnected()) 
197                      mGoogleApiClient.connect(); 
198   
199                  LocationServices.FusedLocationApi.requestLocationUpdates(mGoogleApiClient, mLocationRequest, MainActivity.this); 
200              } 
201          } 
202      }
